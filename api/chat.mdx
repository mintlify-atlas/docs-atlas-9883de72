## Overview

The Chat API enables you to send messages, manage conversations, and execute AI agents. It supports both traditional chat and advanced agent workflows.

## Execute Agent

Execute a single agent with a prompt.

### REST API

```bash
curl -X POST https://your-domain.com/api/agent \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -d '{
    "userId": "user_123",
    "agentId": "agent_456",
    "prompt": "Hello, how can you help me?",
    "autoStart": true
  }'
```

<ParamField path="userId" type="string" required>
  User ID
</ParamField>

<ParamField path="agentId" type="string">
  Agent ID (either agentId or slug required)
</ParamField>

<ParamField path="slug" type="string">
  Agent slug (either agentId or slug required)
</ParamField>

<ParamField path="prompt" type="string" required>
  User prompt/message
</ParamField>

<ParamField path="autoStart" type="boolean">
  Auto-start execution (default: true)
</ParamField>

<ParamField path="appContext" type="object">
  Application context
  
  <ParamField path="appContext.sessionId" type="string">
    Session ID
  </ParamField>
  
  <ParamField path="appContext.topicId" type="string">
    Topic ID
  </ParamField>
  
  <ParamField path="appContext.threadId" type="string">
    Thread ID
  </ParamField>
  
  <ParamField path="appContext.groupId" type="string">
    Group ID
  </ParamField>
</ParamField>

<ParamField path="existingMessageIds" type="array">
  Existing message IDs to include in context
</ParamField>

**Response:**

<ResponseField name="operationId" type="string">
  Operation ID for tracking execution
</ResponseField>

<ResponseField name="autoStarted" type="boolean">
  Whether execution started automatically
</ResponseField>

<ResponseField name="executionTime" type="number">
  Execution time in milliseconds
</ResponseField>

### tRPC API

```typescript
const result = await client.aiAgent.execAgent.mutate({
  agentId: 'agent_123',
  prompt: 'What is the weather today?',
  autoStart: true,
});
```

## Agent Stream (SSE)

Connect to real-time agent execution stream.

```javascript
const eventSource = new EventSource(
  `https://your-domain.com/api/agent/stream?operationId=${operationId}&includeHistory=true`
);

eventSource.addEventListener('message', (event) => {
  const data = JSON.parse(event.data);
  
  switch (data.type) {
    case 'agent_runtime_start':
      console.log('Agent started');
      break;
    case 'agent_step_start':
      console.log('Step started:', data.stepIndex);
      break;
    case 'tool_call':
      console.log('Tool called:', data.tool);
      break;
    case 'agent_runtime_end':
      console.log('Agent completed');
      eventSource.close();
      break;
  }
});
```

<ParamField path="operationId" type="string" required>
  Operation ID from execAgent response
</ParamField>

<ParamField path="lastEventId" type="string" optional>
  Resume from specific event ID (default: '0')
</ParamField>

<ParamField path="includeHistory" type="boolean" optional>
  Include historical events (default: false)
</ParamField>

### Stream Events

<ResponseField name="agent_runtime_start" type="event">
  Agent execution started
  
  <ResponseField name="operationId" type="string">
    Operation ID
  </ResponseField>
  
  <ResponseField name="timestamp" type="number">
    Event timestamp
  </ResponseField>
</ResponseField>

<ResponseField name="agent_step_start" type="event">
  Agent step started
  
  <ResponseField name="stepIndex" type="number">
    Step index
  </ResponseField>
</ResponseField>

<ResponseField name="tool_call" type="event">
  Tool called by agent
  
  <ResponseField name="tool" type="object">
    Tool information
  </ResponseField>
  
  <ResponseField name="arguments" type="object">
    Tool arguments
  </ResponseField>
</ResponseField>

<ResponseField name="tool_result" type="event">
  Tool execution result
  
  <ResponseField name="result" type="any">
    Tool result
  </ResponseField>
</ResponseField>

<ResponseField name="agent_runtime_end" type="event">
  Agent execution completed
  
  <ResponseField name="status" type="string">
    Completion status ('done', 'error', 'interrupted')
  </ResponseField>
</ResponseField>

<ResponseField name="heartbeat" type="event">
  Keep-alive heartbeat (every 30 seconds)
</ResponseField>
