## State Management

### Zustand 5

**Lightweight state management** with slice pattern:

```typescript
import { create } from 'zustand';
import { devtools, persist } from 'zustand/middleware';

interface ChatStore {
  messages: Message[];
  addMessage: (msg: Message) => void;
}

export const useChatStore = create<ChatStore>()()
  devtools(
    persist(
      (set) => ({
        messages: [],
        addMessage: (msg) => set((state) => ({
          messages: [...state.messages, msg]
        })),
      }),
      { name: 'chat-storage' }
    )
  );
```

Stores location: `src/store/`

### Selectors

```typescript
// In store slice
export const chatSelectors = {
  currentMessages: (s: ChatStore) => s.messages.filter(m => m.sessionId === s.activeSession),
};

// Usage
const messages = useChatStore(chatSelectors.currentMessages);
```

## Database

### Drizzle ORM 0.45

**Type-safe SQL ORM:**

```typescript
// Define schema
export const agents = pgTable('agents', {
  id: text('id').primaryKey(),
  name: text('name').notNull(),
  userId: text('user_id').references(() => users.id),
  createdAt: timestamp('created_at').defaultNow(),
});

// Query
const result = await db
  .select()
  .from(agents)
  .where(eq(agents.userId, userId))
  .orderBy(desc(agents.createdAt));
```

Schemas: `packages/database/src/schemas/`
